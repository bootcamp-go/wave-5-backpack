Crea un error personalizado con un struct que implemente “Error()” con el mensaje “error: el salario ingresado no alcanza el mínimo imponible"
y lánzalo en caso de que “salary” sea menor a 150.000. Caso contrario, imprime por consola el mensaje “Debe pagar impuesto”.
*/

package main

import (
	"fmt"
	"os"
)

type errorcito struct {
	status int
	msg    string
}

func (e *errorcito) Error() string {
	fmt.Printf("error: el salario ingresado (%v) no alcanza el mínimo imponible \n", e.status)
	return e.msg
}
func myError(salario int) (int, error) {
	if salario < 150000 {
		return 123, &errorcito{
			status: salario,
			msg:    "algo salió mal",
		}
	}
	return 200, nil
}

func main() {
	salary := 50
	status, err := myError(salary)

	if err != nil {
		fmt.Println(err)
		os.Exit(1)
	}

	fmt.Printf("Status %d, Funciona!", status)

}
