package main

import "fmt"

func impuesto(salario float64) float64 {
	var nuevoSueldo float64
	if salario > 50000 {
		nuevoSueldo = salario - salario*0.17
		return nuevoSueldo
	} else if salario > 150000 {
		nuevoSueldo = salario - salario*0.27
		return nuevoSueldo
	} else {
		return nuevoSueldo
	}
}

func main() {

	r := impuesto(60000)
	fmt.Println("Su sueldo es: ", r)
}
