/*
Una empresa de redes sociales requiere implementar una estructura usuario con funciones que vayan agregando información a la estructura. Para optimizar y ahorrar memoria requieren que
la estructura usuarios ocupe el mismo lugar en memoria para el main del programa y para las funciones:
La estructura debe tener los campos: Nombre, Apellido, edad, correo y contraseña
Y deben implementarse las funciones:
cambiar nombre: me permite cambiar el nombre y apellido.
cambiar edad: me permite cambiar la edad.
cambiar correo: me permite cambiar el correo.
cambiar contraseña: me permite cambiar la contraseña.
*/

package main

import (
	"fmt"
)

type usuario struct {
	Nombre     string
	Apellido   string
	Edad       int
	Coreeo     string
	Contraseña string
}

func (user *usuario) cambiarNombre(name, lastname string) {
	user.Nombre = name
	user.Apellido = lastname

}

func (user *usuario) cambiarEdad(age int) {
	user.Edad = age
}

func (user *usuario) cambiarCorreo(mail string) {
	user.Coreeo = mail
}

func (user *usuario) cambiarContraseña(pass string) {
	user.Contraseña = pass
}

func main() {
	u1 := usuario{"juan", "perez", 15, "hola@hola.com", "asdf"}

	fmt.Println(u1.Edad)
	u1.cambiarEdad(30)
	fmt.Println(u1.Edad)

}
